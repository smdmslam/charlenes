javascriptfunction importTypeformData() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var rawSheet = ss.getSheetByName('RAW_DATA');
  
  // Typeform API credentials (store in Script Properties)
  var apiKey = PropertiesService.getScriptProperties().getProperty('TYPEFORM_API_KEY');
  var formId = PropertiesService.getScriptProperties().getProperty('TYPEFORM_FORM_ID');
  
  var url = 'https://api.typeform.com/forms/' + formId + '/responses';
  
  var options = {
    'method': 'get',
    'headers': {
      'Authorization': 'Bearer ' + apiKey
    }
  };
  
  var response = UrlFetchApp.fetch(url, options);
  var data = JSON.parse(response.getContentText());
  
  // Transform Typeform JSON to spreadsheet rows
  var rows = [];
  data.items.forEach(function(item) {
    var row = [
      item.submitted_at,
      item.token,
      // Map each answer to correct column based on field ID
      // This requires knowing your Typeform field IDs
    ];
    rows.push(row);
  });
  
  // Append to RAW_DATA (skip row 1 which is headers)
  if (rows.length > 0) {
    rawSheet.getRange(rawSheet.getLastRow() + 1, 1, rows.length, rows[0].length).setValues(rows);
  }
  
  Logger.log('Imported ' + rows.length + ' new responses');
}
